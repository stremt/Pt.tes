🔥 FALLBACK JS SUMMARIZER
=========================

If browser does NOT support the AI Summarizer API, implement fallback:

```js
function fallbackSummarize(text) {
    const sentences = text.match(/[^\.!\?]+[\.!\?]+/g) || [text];
    const wordFreq = {};

    const words = text.toLowerCase().match(/\w+/g) || [];
    words.forEach(w => wordFreq[w] = (wordFreq[w] || 0) + 1);

    const scored = sentences.map(s => {
        const score = s.toLowerCase().split(/\W+/).reduce((a, w) => a + (wordFreq[w] || 0), 0);
        return { sentence: s, score };
    });

    scored.sort((a, b) => b.score - a.score);
    
    const limit = Math.max(1, Math.floor(sentences.length * 0.3));
    return scored.slice(0, limit).map(s => s.sentence).join(" ");
}
```
