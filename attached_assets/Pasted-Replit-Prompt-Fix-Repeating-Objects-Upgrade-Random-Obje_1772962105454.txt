1. Implement a Non-Repeating Random Algorithm

Replace the current Math.random() logic with a shuffle-based system.

Logic

When the page loads, create a copy of the object dataset.

Shuffle the entire array using the Fisher-Yates shuffle algorithm.

Store the shuffled list.

Each time the user clicks Generate, pull the next object from the shuffled list.

Continue until the list is exhausted.

When all objects are used, reshuffle the dataset again.

This guarantees that all objects appear once before any repetition.

Example pseudocode:

let objects = [...dataset];
let shuffledObjects = [];
let index = 0;

function shuffleArray(array) {
  for (let i = array.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [array[i], array[j]] = [array[j], array[i]];
  }
}

function initGenerator() {
  shuffledObjects = [...objects];
  shuffleArray(shuffledObjects);
  index = 0;
}

function getRandomObject() {
  if (index >= shuffledObjects.length) {
    initGenerator();
  }
  return shuffledObjects[index++];
}

initGenerator();
