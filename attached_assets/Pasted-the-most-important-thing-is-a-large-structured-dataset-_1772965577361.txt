7️⃣ Username Templates (Important)

Templates combine everything.

const templates = [
"{name}{word}",
"{word}{name}",
"{adj}{name}",
"{name}{noun}",
"{name}_{noun}",
"{adj}_{name}",
"{name}{number}",
"{adj}{noun}{number}",
"{name}{adj}",
"{noun}{name}"
];
8️⃣ Generation Logic

Example system:

function generateGamingUsername(name, gender){

let wordPool = neutralWords;

if(gender === "male") wordPool = maleWords;
if(gender === "female") wordPool = femaleWords;

const noun = gamingNouns[Math.floor(Math.random()*gamingNouns.length)];
const adj = gamingAdjectives[Math.floor(Math.random()*gamingAdjectives.length)];
const word = wordPool[Math.floor(Math.random()*wordPool.length)];
const number = numbers[Math.floor(Math.random()*numbers.length)];

let username = name + noun;

return username;

}
